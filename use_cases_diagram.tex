% ============================================================
%  MACRO VẼ STICKMAN ACTOR
% ============================================================
\def\stickman#1#2{%
    \node[circle, draw=black, thick, minimum size=0.55cm, fill=white]
        (head#1) at (#2) {};
    \coordinate (body#1) at ($(head#1.south)-(0,0.65)$);
    \draw[thick] (head#1.south) -- (body#1);
    % arms
    \draw[thick]
        ($(head#1.south)!0.33!(body#1)-(0.5,0)$) --
        ($(head#1.south)!0.33!(body#1)+(0.5,0)$);
    % legs
    \draw[thick] (body#1) -- ++(-0.4,-0.6);
    \draw[thick] (body#1) -- ++(0.4,-0.6);
}
